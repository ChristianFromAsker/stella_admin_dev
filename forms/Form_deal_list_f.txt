Private Sub fix_status_buttons(ByVal status_input As Variant, ByVal status_buttons As Variant)
    Load.check_conn_and_variables
    'For status_buttons:
    '0 = control name
    '1 = default caption
    '2 = control type
    '3 = caption for on
    '4 = caption for off
    Dim i As Integer: i = 0
    For i = 1 To UBound(status_input)
        With Me.Controls(status_buttons(i, 0))
            If status_input(i) = 1 Then
                .Caption = status_buttons(i, 3)
                .BackColor = Load.colors.active_filter_bg
                .ForeColor = colors.white
            Else
                .Caption = status_buttons(i, 4)
                .BackColor = Load.colors.inactive_filter_bg
                .ForeColor = colors.black
            End If
        End With
    Next i
End Sub

Private Sub comSector_AfterUpdate()
    Load.check_conn_and_variables
    If Me!comSector = 111 Then GoTo outro ' 111 is _All
        
    'populate com_sub_sector based on selection
    Dim str_sql As String, rs As ADODB.Recordset
    'remove existing items
    With Me!com_sub_sector
        Do While .ListCount > 0
            .RemoveItem (0)
        Loop
        .AddItem "-1;_All"
        str_sql = "SELECT sector_id id, sector_name menu_item FROM " & sources.sectors_table & " WHERE parent_sector_id = " & Me!comSector & " ORDER BY sector_name"
        Set rs = utilities.create_adodb_rs(conn, str_sql)
            Do While rs.EOF = False
                .AddItem rs!id & ";" & rs!menu_item
                rs.MoveNext
            Loop
        rs.Close
        .Value = -1
    End With
outro:
    If Not rs Is Nothing Then
        If rs.State = 1 Then rs.Close
        Set rs = Nothing
    End If
    Exit Sub

        & "Error number: " & Err.Number & vbNewLine _

Private Sub cmdRefresh_Click()
    'intro
    Load.check_conn_and_variables
    
    Dim strRPEntityCondition As String, strStatusCondition As String, strSectorCondition As String, strDateCondition As String, strBrokerCondition As String
    Dim str_has_fee_condition As String, str_fee_paid_condition As String, strJurisdictionCondition As String
    Dim strCondition As String
    Dim y As Integer
    Dim intMonthEnd As Integer, intYearEnd As Integer, intDayEnd As Integer
    Dim int_color_yes As Variant, int_color_no As Variant
    ' RP entity String builder
    strRPEntityCondition = ""
    strRPEntityCondition = EntitySelector(strRPEntityCondition, 86, "cmdAsiaPac")
    strRPEntityCondition = EntitySelector(strRPEntityCondition, 76, "cmdDenmark")
    strRPEntityCondition = EntitySelector(strRPEntityCondition, 82, "cmdFinland")
    strRPEntityCondition = EntitySelector(strRPEntityCondition, 79, "cmdGermany")
    strRPEntityCondition = EntitySelector(strRPEntityCondition, 83, "cmdHolland")
    strRPEntityCondition = EntitySelector(strRPEntityCondition, 87, "cmdMed")
    strRPEntityCondition = EntitySelector(strRPEntityCondition, 77, "cmdNorway")
    strRPEntityCondition = EntitySelector(strRPEntityCondition, 78, "cmdSweden")
    strRPEntityCondition = EntitySelector(strRPEntityCondition, 81, "cmdUK")
    strRPEntityCondition = EntitySelector(strRPEntityCondition, 435, "cmd_canada")
    strRPEntityCondition = EntitySelector(strRPEntityCondition, 436, "cmd_dubai")
    strRPEntityCondition = EntitySelector(strRPEntityCondition, 93, "cmd_usa")
    If strRPEntityCondition = "" Then
        MsgBox "You must choose at least one RP entity."
        Exit Sub
    End If
    ' Status string builder
    strStatusCondition = ""
    strStatusCondition = StatusSelector(strStatusCondition, 1, "cmdNDA")
    strStatusCondition = StatusSelector(strStatusCondition, 2, "cmdNBI")
    strStatusCondition = StatusSelector(strStatusCondition, 481, "cmd_submission")
    strStatusCondition = StatusSelector(strStatusCondition, 3, "cmdPreferred")
    strStatusCondition = StatusSelector(strStatusCondition, 4, "cmdExpensed")
    strStatusCondition = StatusSelector(strStatusCondition, 5, "cmdUW")
    strStatusCondition = StatusSelector(strStatusCondition, 6, "cmdSigned")
    strStatusCondition = StatusSelector(strStatusCondition, 436, "cmd_closed")
    strStatusCondition = StatusSelector(strStatusCondition, 7, "cmdDeclined")
    strStatusCondition = StatusSelector(strStatusCondition, 8, "cmdCancel")
    strStatusCondition = StatusSelector(strStatusCondition, 9, "cmdLost")
    strStatusCondition = StatusSelector(strStatusCondition, 485, "cmd_collapsed")
    If strStatusCondition = "" Then
        MsgBox "You must choose at least one deal status."
        Exit Sub
    End If
    
    ' Date string builder
    Dim date_stop As Boolean
    date_stop = False
    'Check that date input is numbers
    If IsNumeric(txtYearEnd) = False Or IsNumeric(txtYearStart) = False Or IsNumeric(txtMonthStart) = False Or IsNumeric(txtMonthEnd) = False Then
        date_stop = True
    Else
        Dim month_start As Long, month_end As Long, year_start As Long, year_end As Long
        month_start = Me!txtMonthStart: month_end = Me!txtMonthEnd: year_start = Me!txtYearStart: year_end = Me!txtYearEnd
        'check that years are in acceptable intervals
        If year_end > Year(Date) + 1 Or year_start > Year(Date) + 1 Or year_start < 2010 Or year_end < 2010 Then
            date_stop = True
        End If
        If month_start > 12 Or month_end > 12 Then
            date_stop = True
        End If
        'check that start date is earlier than end date
        If year_start > year_end Or (year_start = year_end And month_start > month_end) Then
            date_stop = True
        End If
    End If
    If date_stop = True Then
        MsgBox "Something is wrong with the date. Please make sure the date fields have numbers and the 'from' date is before the 'to' date", , "Issue with dates"
        Me!txtYearEnd = Year(Date): Me!txtYearStart = Year(Date): Me!txtMonthStart = 1: Me!txtMonthEnd = 12
        Exit Sub
    End If
    intDayEnd = 1
    If month_end = 12 Then
        intMonthEnd = 1
        intYearEnd = txtYearEnd + 1
    Else
        intMonthEnd = txtMonthEnd + 1
        intYearEnd = txtYearEnd
    End If
    Dim date_start As Date
    Dim date_end As Date
    date_start = "1-" & month_start & "-" & year_start
    date_end = intDayEnd & "-" & intMonthEnd & "-" & intYearEnd
    If InStr(cmd_date_filter.Caption, "nda") = 0 Then
        strDateCondition = " AND (inception_date >= '" & utilities.generate_sql_date(date_start) & "') AND (inception_date < '" & utilities.generate_sql_date(date_end) & "')"
    Else
        strDateCondition = " AND (create_date >= '" & utilities.generate_sql_date(date_start) & "') AND (create_date < '" & utilities.generate_sql_date(date_end) & "')"
    End If
    
    ' Broker builder condition
    Dim strBrokerSearch As String
    strBrokerSearch = comBroker
    If Me!comBroker = -1 Then
        strBrokerCondition = ""
    Else
        strBrokerCondition = " AND broker_firm_id = " & comBroker
    End If
    
    ' Risk type string builder
    Dim str_risk_type_selection As String, int_risk_type As Long
    If com_risk_type = -1 Then
        str_risk_type_selection = ""
    Else
        str_risk_type_selection = " AND (risk_type_id = " & com_risk_type & ")"
        int_risk_type = com_risk_type
    End If
    
    'Super sector string builder
    If comSector = -1 Then
        strSectorCondition = ""
    Else
        strSectorCondition = " AND (target_super_sector_id = " & Me!comSector & ")"
    End If
    
    'Sub sector string builder
    Dim sub_sector_condition As String
    If Me!com_sub_sector = -1 Then
        sub_sector_condition = ""
    Else
        sub_sector_condition = " AND (target_sub_sector_id = " & Me!com_sub_sector & ")"
    End If
    
    ' Layer type string builder
    Dim str_layer_condition
    If com_primary_or_xs = -1 Then
        str_layer_condition = ""
    Else
        str_layer_condition = " AND primary_or_xs_id = " & com_primary_or_xs
    End If
    
    ' SPA law string builder
    Dim strJurisdictionSelection As String
    If comJurisdiction = "_All" Then
        strJurisdictionCondition = ""
        strJurisdictionSelection = "_All"
    Else
        strJurisdictionCondition = " AND (spa_law = '" & comJurisdiction & "')"
        strJurisdictionSelection = comJurisdiction
    End If
    
    'deal name string builder
    Dim deal_name_condition As String
    deal_name_condition = ""
    If Me!deal_name_search <> "" Then
        deal_name_condition = " AND deal_name LIKE '%" & deal_name_search & "%'"
    End If
    
    ' Final string builder
    strCondition = "(" & strRPEntityCondition & ")" & " AND (" & strStatusCondition & ")" & strDateCondition & strBrokerCondition & strSectorCondition & strJurisdictionCondition _
        & str_risk_type_selection & str_layer_condition & deal_name_condition & sub_sector_condition
    Central.update_deal_list_f strCondition
End Sub
Function EntitySelector(ByRef strExisting As String, ByVal intEntity As Integer, ByVal strButton As String)
    Load.check_conn_and_variables
    Dim strAddition As String
    
    If Controls(strButton).BackColor = Load.colors.active_filter_bg Then
        If strExisting <> "" Then
            strAddition = " OR budget_home_id = " & intEntity
        Else
            strAddition = " budget_home_id = " & intEntity
        End If
    End If
    EntitySelector = strExisting & strAddition
End Function
Function StatusSelector(ByRef strExisting As String, ByVal intStatus As Integer, ByVal strButton As String)
    Load.check_conn_and_variables
    Dim strAddition As String
    If Controls(strButton).BackColor = Load.colors.active_filter_bg Then
        If strExisting <> "" Then
            strAddition = " OR "
        Else
            strAddition = " "
        End If
        Dim str_status_text As String
        str_status_text = "deal_status_id = " & intStatus
        strAddition = strAddition & str_status_text
        Dim intTracker
        intTracker = "1"
    Else
    End If
    StatusSelector = strExisting & strAddition
End Function
Private Sub fix_deal_list_order(ByVal data_field As String)
    Load.check_conn_and_variables
    Dim str_sql() As String
    str_sql = Split(Me.RecordSource, "WHERE")
    str_sql = Split(str_sql(1), "ORDER")
    Dim str_direction As String
    Dim str_control As String
    str_control = "sort_" & data_field
    If Me.Controls(str_control).Caption = "<<<" Then
        Me.Controls(str_control).Caption = ">>>"
        str_direction = "ASC"
    Else
        Me.Controls(str_control).Caption = "<<<"
        str_direction = "DESC"
    End If
    open_forms.deal_list_f
    Central.update_deal_list_f str_sql(0) & " ORDER BY " & data_field & " " & str_direction
End Sub
