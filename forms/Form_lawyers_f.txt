Private Sub cmd_change_Click()
    Dim proc_name As String
    proc_name = "lawyers_f.cmd_change.Click"
    Load.call_stack = proc_name
    On Error GoTo err_handler
    If Load.is_debugging = True Then On Error GoTo 0
    
    Load.check_conn_and_variables
    
    Dim arr_fields() As Variant
    Dim i As Integer
    Dim str_form As String
    
    arr_fields = declare_fields
    'Populate change fields with data
    With Me
        For i = 1 To arr_fields(0, 0)
            .Controls("new_" & arr_fields(i, 0)) = .Controls(arr_fields(i, 0))
            If arr_fields(i, 1) = 1 And IsNull(.Controls(arr_fields(i, 0))) Then
                .Controls(arr_fields(i, 0)).BackColor = colors.yellow
            End If
        Next i
    End With
    Me!cmd_add.Caption = "Change lawyer"
outro:
    Exit Sub
err_handler:
    Dim err_object As cls_err_object
    Set err_object = New cls_err_object
    With err_object
        .routine_name = proc_name
        .milestone = ""
        .params = "me!new_lawyer_id = " & Nz(Me!new_lawyer_id, "not provided")
        .system_error_code = Err.Number
        .system_error_text = Err.Description
        .show_error_msg = True
        .send_error err_object
    End With
    GoTo outro
End Sub
Private Sub cmd_add_Click()
    Dim proc_name As String
    proc_name = "lawyers_f.cmd_add.Click"
    On Error GoTo err_handler
    If Load.is_debugging = True Then On Error GoTo 0
    
    Load.check_conn_and_variables
    
    Dim arr_fields() As Variant
    Dim boo_cancel As Boolean
    Dim str_sql As String
    Dim rs As ADODB.Recordset
    Dim i As Integer

    'Check that necessary fields are populated
    arr_fields = declare_fields
    
    boo_cancel = False
    For i = 1 To arr_fields(0, 0)
        If arr_fields(i, 1) = 1 And IsNull(Me.Controls("new_" & arr_fields(i, 0))) Then
            Me.Controls("new_" & arr_fields(i, 0)).BackColor = colors.yellow
            boo_cancel = True
        End If
    Next i
    If boo_cancel = True Then
        Exit Sub
    End If
    Dim id As Long
    id = -1 'pending new approach
    If IsNull(Me!new_lawyer_id) Or Me!new_lawyer_id = "" Then
        str_sql = "INSERT INTO " & sources.lawyers_table & " (personal_name, is_counsel, law_firm, Jurisdiction, Comments) VALUES (" _
            & "'" & Me!new_personal_name & "'" _
            & ", " & Me!new_is_counsel _
            & ", " & Me!new_law_firm _
            & ", '" & Me!new_jurisdiction & "'" _
            & ", '" & Me!new_comments & "'" _
            & ")"
    Else
        str_sql = "UPDATE " & sources.lawyers_table & " SET " _
            & "personal_name = '" & Me!new_personal_name & "'" _
            & ", law_firm = " & Me!new_law_firm _
            & ", is_counsel = " & Me!new_is_counsel _
            & ", Jurisdiction = '" & Me!new_jurisdiction & "'" _
            & ", Comments = '" & Me!new_comments & "'" _
            & " WHERE lawyer_id = " & Me!new_lawyer_id & ";"
            
            id = Me!new_lawyer_id
    End If
    conn.Execute str_sql
    'Update recordsource
    fix_rs.lawyers_f
    'Reset form
    With Me
        For i = 1 To arr_fields(0, 0)
            .Controls("new_" & arr_fields(i, 0)) = ""
            .Controls("new_" & arr_fields(i, 0)).BackColor = colors.white
        Next i
    End With
    Me!cmd_add.Caption = "add"
outro:
    If Not rs Is Nothing Then
        If rs.State = 1 Then rs.Close
        Set rs = Nothing
    End If
    Exit Sub
err_handler:
    MsgBox "Something went wrong. Try once more. If it fails again, snip this to Christian." & vbNewLine & vbNewLine _
        & "Error number: " & Err.Number & vbNewLine _
        & "Error description: " & Err.Description & vbNewLine _
        & "Where: lawyers_f.cmd_add_change_Click" & vbNewLine _
        & "Parameters: id = " & id & vbNewLine _
        & "App: Stella Admin Eur", , " Whoopsie daisies (like Hugh Grant in Notting Hill)"
    GoTo outro
End Sub
Private Sub cmd_delete_Click()
    On Error GoTo err_handler
    Load.check_conn_and_variables

    Dim str_message As String
    Dim str_name As String

    str_message = "You are about to delete " & str_name & ". Please confirm."
    Dim answer As Variant
    answer = MsgBox(str_message, vbQuestion + vbYesNo, "Confirm deletion.")

        Exit Sub
    Else

outro:

        Set rs = Nothing
    End If

End Sub

Private Function declare_fields()

    Dim arr_fields(0 To 20, 0 To 1) As Variant
    Dim i As Integer
    
    'For the second value, 0 is the field name and 1 is 0 if optional and 1 if mandatory.
    i = 1
    arr_fields(i, 0) = "personal_name"
    arr_fields(i, 1) = 1
    
    i = i + 1
    arr_fields(i, 0) = "is_counsel"
    arr_fields(i, 1) = 1
    
    i = i + 1
    arr_fields(i, 0) = "law_firm"
    arr_fields(i, 1) = 1
    
    i = i + 1
    arr_fields(i, 0) = "jurisdiction"
    arr_fields(i, 1) = 1
    
    i = i + 1
    arr_fields(i, 0) = "comments"
    arr_fields(i, 1) = 0
    
    i = i + 1
    arr_fields(i, 0) = "lawyer_id"
    arr_fields(i, 1) = 0
    
    arr_fields(0, 0) = i
    
    declare_fields = arr_fields
End Function
