Private Sub cmd_change_Click()
    On Error GoTo err_handler
    If Load.is_debugging = True Then On Error GoTo 0
    
    Load.check_conn_and_variables
    Dim str_sql As String, rs As ADODB.Recordset
    Dim str_recordset As String
    str_recordset = Load.sources.broker_firms_view
    
    Dim input_controls() As Variant, i As Integer
    input_controls = declare_fields_local
    
    str_sql = "SELECT * FROM " & str_recordset & " WHERE broker_firm_id = " & Me!id
    Set rs = utilities.create_adodb_rs(conn, str_sql)
        With Me
            For i = 1 To input_controls(0, 0)
                If IsNull(rs.Fields(input_controls(i, 0))) = False Then
                    .Controls("new_" & input_controls(i, 0)) = rs.Fields(input_controls(i, 0))
                End If
            Next i
        End With
    rs.Close
    Me!cmd_add_change.Caption = "update"
    
outro:
    If Not rs Is Nothing Then
        If rs.State = 1 Then rs.Close
        Set rs = Nothing
    End If
    Exit Sub
    
err_handler:
    Dim err_object As cls_err_object
    Set err_object = New cls_err_object
    With err_object
        .routine_name = "broker_firms_f.cmd_change.Click"
        .milestone = ""
        .params = "str_sql = " & str_sql
        .system_error_code = Err.Number
        .system_error_text = Err.Description
        .show_error_msg = True
        .send_error err_object
    End With
    
    GoTo outro

End Sub
Private Sub cmd_add_change_Click()
    On Error GoTo err_handler
    If Load.is_debugging = True Then On Error GoTo 0
    Load.check_conn_and_variables
    
    Dim str_sql As String, rs As ADODB.Recordset, str_recordset As String
    str_recordset = Load.sources.broker_firms_table
    
    'Check that necessary fields are populated
    Dim arr_fields() As Variant
    arr_fields = declare_fields_local
    Dim boo_cancel As Boolean
    boo_cancel = False
    Dim i As Integer
    With global_vars.control_list_propertes
        For i = 1 To arr_fields(0, 0)
            If arr_fields(i, .is_mandatory) = True _
                And (IsNull(Me.Controls("new_" & arr_fields(i, .field_name))) _
                Or IsEmpty(Me.Controls(.field_name)) = True Or Me.Controls(.field_name) = "" _
               ) Then
                Me.Controls("new_" & arr_fields(i, .field_name)).BackColor = colors.yellow
                boo_cancel = True
            End If
        Next i
    End With
    
    If boo_cancel = True Then
        GoTo outro
    End If
    
    'Insert data into table
    'first, get id. If new, create new record
    Dim id As Long
    If IsNull(Me!new_broker_firm_id) = True Or Me!new_broker_firm_id = "" Then
        'check if broker is already in list
        str_sql = "SELECT broker_firm_id, business_name FROM " & Load.sources.broker_firms_table & " WHERE is_deleted = 0 AND business_name = '" & Me!new_business_name & "'"
        Set rs = utilities.create_adodb_rs(conn, str_sql)
            If rs.RecordCount > 0 Then
                MsgBox "Broker with that business name already exist in the list.", , "Duplicate"
                GoTo outro
            End If
        rs.Close
            
        'assuming broker was not in list, insert new id
        str_sql = "INSERT INTO " & str_recordset & "(is_deleted) VALUES (0)"
        conn.Execute str_sql, adExecuteNoRecords
        str_sql = "SELECT broker_firm_id FROM " & str_recordset & " ORDER BY broker_firm_id DESC"
        Set rs = utilities.create_adodb_rs(conn, str_sql)
            id = rs!broker_firm_id
        rs.Close: Set rs = Nothing
    Else
        id = Me!new_broker_firm_id
    End If
    
    Dim jurisdiction_data As String
    jurisdiction_data = ""
    If IsNull(Me!new_jurisdiction_id) = False And Me!new_jurisdiction_id <> "" Then
        jurisdiction_data = ", jurisdiction_id__jurisdictions_t = " & Me!new_jurisdiction_id
    End If
    
     'Update data
    str_sql = "UPDATE " & str_recordset & " SET " _
        & "business_name = '" & Me!new_business_name & "'" _
        & ", legal_name = '" & Me!new_legal_name & "'" _
        & ", broker_street = '" & Me!new_broker_street & "'" _
        & ", broker_zip = '" & Me!new_broker_zip & "'" _
        & ", broker_city = '" & Me!new_broker_city & "'" _
        & ", region = '" & Me!new_region & "'" _
        & jurisdiction_data _
        & ", license_no = '" & Me!new_license_no & "'" _
        & ", brand = '" & Me!new_brand & "'" _
        & " WHERE broker_firm_id = " & id
    conn.Execute str_sql, adExecuteNoRecords
    Me!cmd_add_change.Caption = "add"
    
    fix_rs.broker_firms_f
    
    'empty input fields
    With Me
        For i = 1 To arr_fields(0, 0)
            .Controls("new_" & arr_fields(i, 0)) = ""
            If arr_fields(i, global_vars.control_list_propertes.set_bg_to_white) = True Then
                .Controls("new_" & arr_fields(i, 0)).BackColor = colors.white
            End If
        Next i
    End With
    
outro:
    If Not rs Is Nothing Then
        If rs.State = 1 Then rs.Close
        Set rs = Nothing
    End If
    Exit Sub
    
err_handler:
    Dim err_object As cls_err_object
    Set err_object = New cls_err_object
    With err_object
        .routine_name = "broker_firms_f.cmd_add_change.Click"
        .milestone = ""
        .params = "str_sql = " & str_sql
        .system_error_code = Err.Number
        .system_error_text = Err.Description
        .show_error_msg = True
        .send_error err_object
    End With

End Sub
Private Sub cmd_delete_Click()
    Dim str_message As String
    Dim str_firm_name As String
    str_firm_name = Me!short_name
    str_message = "You are about to delete " & str_firm_name & ". Please confirm."
    Dim answer As Variant
    answer = MsgBox(str_message, vbQuestion + vbYesNo, "Confirm deletion.")
    If answer = vbNo Then
        Exit Sub
    Else
        conn.Execute "UPDATE " & sources.broker_firms_table & " SET is_deleted = 1 WHERE broker_firm_id = " & Me!id
        fix_rs.broker_firms_f
    End If
End Sub
