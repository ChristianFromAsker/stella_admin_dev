Private Sub cmd_update_Click()
    Load.check_conn_and_variables
    Dim arr_fields() As Variant
    arr_fields = declare_fields
    For i = 1 To arr_fields(0, 0)
        If IsNull(Me.Controls(arr_fields(i, 0))) And arr_fields(i, 1) = 1 Then
            MsgBox "Please populate all yellow fields"
            Exit Sub
        End If
    Next i
    Dim str_sql As String
    str_sql = "UPDATE " & sources.insurers_table & " SET " _
        & "Name = '" & Me!Name & "'" _
        & ", insurer_type = " & Me!insurer_type _
        & ", is_playing = " & Me!is_playing _
        & " WHERE id = " & Me!id
    conn.Execute str_sql
    MsgBox Me!Name & " was updated. This form will now close.", , "Great success!"
    fix_rs.insurers_f
    DoCmd.Close acForm, Me.Name
End Sub
Private Sub is_playing_AfterUpdate()
    Me!is_playing.BackColor = colors.white
End Sub
Private Sub Form_Load()
    DoCmd.MoveSize Right:=1000, Down:=1000, Width:=10000, Height:=3000
End Sub
Private Sub insurer_type_AfterUpdate()
    Me!insurer_type.BackColor = colors.white
End Sub
Private Function declare_fields()
    Const int_number_of_fields As Integer = 4
    Dim i As Integer: i = 0
    Dim arr_fields(0 To int_number_of_fields, 0 To 1) As Variant
    'For the second value, 0 is the field name and 1 is 0 if optional and 1 if mandatory.
    arr_fields(i, 0) = int_number_of_fields
    arr_fields(i, 1) = 0: i = i + 1
    arr_fields(i, 0) = "name"
    arr_fields(i, 1) = 1: i = i + 1
    arr_fields(i, 0) = "is_playing"
    arr_fields(i, 1) = 1: i = i + 1
    arr_fields(i, 0) = "insurer_type"
    arr_fields(i, 1) = 1: i = i + 1
    arr_fields(i, 0) = "id"
    arr_fields(i, 1) = 0: i = i + 1
    declare_fields = arr_fields
End Function
