Private Sub cmd_set_shown_to_closed_Click()
    Load.call_stack = Me.Name & ".cmd_set_shown_to_closed.Click"
    Load.check_conn_and_variables
    Dim str_where As String, str_sql As String
    
    If MsgBox("Are you sure you want to close these issues?", vbQuestion + vbYesNo + vbDefaultButton2, "Confirm delete") = vbNo Then
        Exit Sub
    End If
        
    str_where = Split(Me.RecordSource, "WHERE")(1)
    str_where = Split(str_where, "ORDER")(0)
    str_sql = "UPDATE " & Load.sources.log_errors_eur_t & " SET is_closed_id = 93 WHERE " & str_where
    conn.Execute str_sql
    str_sql = "UPDATE " & Load.sources.log_errors_us_t & " SET is_closed_id = 93 WHERE " & str_where
    conn.Execute str_sql
    
    global_vars.system_dashboard.filter_routine_name.field_value = "_all"
    
    With global_vars.system_dashboard.filter_routine_name
        Me.Controls(.field_name) = .field_value
    End With
    
    fix_rs.system_dashboard_f
    
outro:
    Load.call_stack = ""
    
End Sub

Private Sub header_routine_name_AfterUpdate()
    Load.call_stack = Me.Name & "." & Screen.ActiveControl.Name
    Load.check_conn_and_variables
    On Error GoTo err_handler
    If Load.is_debugging = True Then On Error GoTo 0
    
    global_vars.system_dashboard.filter_routine_name.field_value = Me!header_routine_name
    fix_rs.system_dashboard_f

outro:
    Load.call_stack = ""
    open_forms.working_on_it_f__close
    Exit Sub

err_handler:
    Dim err_object As cls_err_object
    Set err_object = New cls_err_object
    With err_object
        .routine_name = "system_dashboard_f.header_routine_name.afterupdate"
        .milestone = ""
        .params = ""
        .system_error_code = Err.Number
        .system_error_text = Err.Description
        .show_error_msg = True
        .send_error err_object
    End With
    GoTo outro
End Sub

Private Sub header_txt_input_deal_id_AfterUpdate()
    Const proc_name As String = "deal_log_f.header_txt_input_deal_id.AfterUpdate"
    utilities.call_stack_add_item proc_name
    On Error GoTo err_handler
    If Load.is_debugging = True Then On Error GoTo 0
    
    Load.check_conn_and_variables
    
    fix_rs.deal_log_f Me!header_txt_input_deal_id
outro:
    Load.call_stack = ""
    Exit Sub

err_handler:
    Central.err_handler proc_name, Err.Number, Err.Description, "", "", "", True
    Resume outro
End Sub
