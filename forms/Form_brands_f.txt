Private Sub cmd_show_all_Click()
    With cmd_show_all
        If .Caption = "show inactive" Then
            .Caption = "show only active"
            fix_rs.brands_f False
        Else
            .Caption = "show inactive"
            fix_rs.brands_f True
        End If
    End With
End Sub
Private Sub cmd_cancel_Click()
    DoCmd.Close acForm, Me.Name
End Sub
Private Sub cmd_change_Click()
    'intro
    On Error GoTo err_handler
    Load.check_conn_and_variables

    Dim str_form As String, i As Integer
    Dim arr_fields() As Variant
    arr_fields = declare_fields
    'Populate change fields with data
    Dim str_sql As String
    str_sql = "SELECT brand_id, brand_name, brand_type_id, is_active FROM " & sources.brands_table & " WHERE brand_id = " & Me!brand_id
    Dim rs As ADODB.Recordset
    Set rs = utilities.create_adodb_rs(conn, str_sql)
        With Me
            For i = 1 To arr_fields(0, 0)
                .Controls("new_" & arr_fields(i, 0)) = rs.Fields(arr_fields(i, 0))
            Next i
        End With
    rs.Close
    Set rs = Nothing
    Me!cmd_add.Caption = "update"
    Exit Sub
err_handler:
    MsgBox "Something went wrong. Try once more. If it fails again, snip this to Christian." & vbNewLine & vbNewLine _
        & "Error number: " & Err.Number & vbNewLine _
        & "Error description: " & Err.Description & vbNewLine _
        & "Where: add_uws_f.cmd_change_Click" & vbNewLine _
        & "Parameters: me!id" & Me!id & vbNewLine _
        & "App: Stella Admin Eur", , " Whoopsie daisies (like Hugh Grant in Notting Hill)"

End Sub
Private Sub cmd_add_Click()
    Const proc_name As String = "brands_f.cmd_add.Click"
    utilities.call_stack_add_item proc_name
    On Error GoTo 0
    If Load.is_debugging = True Then On Error GoTo 0
    
    Load.check_conn_and_variables
    
    Dim arr_fields() As Variant
    Dim boo_cancel As Boolean
    Dim i As Integer
    Dim id As Long
    Dim rs As ADODB.Recordset
    Dim str_sql As String
    
    arr_fields = declare_fields
    boo_cancel = False
    
    For i = 1 To arr_fields(0, 0)
        If arr_fields(i, 1) = 1 And IsNull(Me.Controls("header_" & arr_fields(i, 0))) Then
            Me.Controls("header_" & arr_fields(i, 0)).BackColor = colors.yellow
            boo_cancel = True
        End If
    Next i
    
    If boo_cancel = True Then
        GoTo outro
    End If
    
    If IsNull(Me!header_brand_id) = True Or Me!header_brand_id = "" Then
        str_sql = "INSERT INTO " & sources.brands_table & "(is_deleted) VALUES (0)"
        conn.Execute str_sql, adExecuteNoRecords
        
        str_sql = "SELECT brand_id FROM " & sources.brands_table & " ORDER BY brand_id DESC"
        Set rs = utilities.create_adodb_rs(conn, str_sql)
            id = rs!brand_id
        rs.Close
        Set rs = Nothing
    Else
        id = Me!header_brand_id
    End If
    
    str_sql = "UPDATE " & sources.brands_table & " SET " _
    & "brand_name= '" & Me!header_brand_name & "'" _
    & ", is_active = " & Me!header_is_active _
    & ", is_advisor = " & Me!header_is_advisor _
    & ", is_broker = " & Me!header_is_broker _
    & ", is_carrier = " & Me!header_is_carrier _
    & ", is_competitor = " & Me!header_is_competitor _
    & ", is_party = " & Me!header_is_party _
    & " WHERE brand_id = " & id

    conn.Execute str_sql, adExecuteNoRecords
    
    fix_rs.brands_f False
    
    With Me
        For i = 1 To arr_fields(0, 0)
            .Controls("header_" & arr_fields(i, 0)) = ""
            .Controls("header_" & arr_fields(i, 0)).BackColor = colors.white
        Next i
    End With
    Me!cmd_add.Caption = "Add"
outro:
    utilities.call_stack_remove_last_item
    Exit Sub
err_handler:
    Central.err_handler proc_name, Err.Number, Err.Description, "str_sql = " & str_sql, "", "", True
    Resume outro
End Sub
Private Sub cmd_delete_Click()
    Dim str_message As String
    str_message = "You are about to delete " & Me!brand_name & ". Please confirm."
    Dim answer As Variant
    answer = MsgBox(str_message, vbQuestion + vbYesNo, "Confirm deletion.")
    If answer = vbNo Then
        Exit Sub
    Else
        conn.Execute "UPDATE " & sources.brands_table & " SET is_deleted = 1 WHERE brand_id = " & Me!brand_id, adExecuteNoRecords
        fix_rs.brands_f False
    End If
End Sub
Private Function declare_fields()
    Dim i As Integer, arr_fields(0 To 30, 0 To 1) As Variant
    'For the second value, 0 is the field name and 1 is 0 if optional and 1 if mandatory.
    i = 1
    arr_fields(1, 0) = "brand_name"
    arr_fields(i, 1) = 1
    
    i = i + 1
    arr_fields(i, 0) = "is_active"
    arr_fields(i, 1) = 1
    
    i = i + 1
    arr_fields(i, 0) = "is_advisor"
    arr_fields(i, 1) = 1
    
    i = i + 1
    arr_fields(i, 0) = "is_broker"
    arr_fields(i, 1) = 1
    
    i = i + 1
    arr_fields(i, 0) = "is_carrier"
    arr_fields(i, 1) = 1
    
    i = i + 1
    arr_fields(i, 0) = "is_competitor"
    arr_fields(i, 1) = 1
    
    i = i + 1
    arr_fields(i, 0) = "is_party"
    arr_fields(i, 1) = 1
    
    arr_fields(0, 0) = i
    declare_fields = arr_fields
End Function
