Private Sub cmd_change_Click()
    Load.call_stack = "cmd_change_Click"
    Load.check_conn_and_variables
    On Error GoTo err_handler
    If Load.is_debugging = True Then On Error GoTo 0
    Dim str_sql As String, rs As ADODB.Recordset
    
    str_sql = "SELECT id, PersonalName, BrokerEmail, BrokerFirm, broker_title FROM " & Load.sources.broker_persons_table & " WHERE id = " & Me!broker_person_id
    Set rs = utilities.create_adodb_rs(conn, str_sql)
    With rs
        Me!new_personal_name = !PersonalName
        Me!new_broker_title = !broker_title
        Me!new_broker_firm_id = !BrokerFirm
        Me!new_broker_email = !BrokerEmail
        Me!new_broker_person_id = Me!broker_person_id
        .Close
    End With
outro:
    Load.call_stack = ""
    Exit Sub
    
err_handler:
    Dim err_object As cls_err_object
    Set err_object = New cls_err_object
    With err_object
        .call_stack = Load.call_stack
        .routine_name = "broker_persons_f.cmd_change.Click"
        .milestone = ""
        .params = ""
        .system_error_code = Err.Number
        .system_error_text = Err.Description
        .show_error_msg = True
        .send_error err_object
    End With
    GoTo outro
End Sub
Private Sub cmd_add_update_Click()
    Load.call_stack = "broker_persons_f.cmd_close_save"
    Load.check_conn_and_variables
    On Error GoTo err_handler
    If Load.is_debugging = True Then On Error GoTo 0
    
    Dim str_sql As String, rs As ADODB.Recordset, i As Integer

    Dim str_form As String
    str_form = "broker_person_f"
    
    'Check that necessary fields are populated
    Dim arr_fields() As Variant
    arr_fields = declare_fields_local
    Dim boo_cancel As Boolean
    boo_cancel = False
    For i = 1 To arr_fields(0, 0)
        If arr_fields(i, global_vars.control_list_propertes.is_mandatory) = True And IsNull(Me.Controls("new_" & arr_fields(i, 0))) Then
            Me.Controls("new_" & arr_fields(i, 0)).BackColor = colors.yellow
            boo_cancel = True
        End If
    Next i
    If boo_cancel = True Then
        GoTo outro
    End If
    
    'Insert data into table
    'first, get id. If new, create new record
    Dim broker_person_id As Long
    If IsNull(Me!new_broker_person_id) = True Or Me!new_broker_person_id = "" Then
        'check if broker person is already in list
        str_sql = "SELECT broker_person_id, personal_name FROM " & Load.sources.broker_persons_view & " WHERE personal_name = '" & Me!new_personal_name & "'"
        Set rs = utilities.create_adodb_rs(conn, str_sql)
            If rs.RecordCount > 0 Then
                MsgBox "Broker with that name already exist in the list.", , "Duplicate"
                GoTo outro
            End If
        rs.Close
            
        'assuming broker was not in list, insert new id
        str_sql = "INSERT INTO " & Load.sources.broker_persons_table & " (is_deleted) VALUES (0)"
        conn.Execute str_sql, adExecuteNoRecords
        str_sql = "SELECT id FROM " & Load.sources.broker_persons_table & " ORDER BY id DESC LIMIT 1"
        Set rs = utilities.create_adodb_rs(conn, str_sql)
            broker_person_id = rs!id
        rs.Close
        Set rs = Nothing
    Else
        broker_person_id = Me!new_broker_person_id
    End If
    
     'Update data
    str_sql = "UPDATE " & Load.sources.broker_persons_table & " SET " _
        & "PersonalName = '" & Me!new_personal_name & "'" _
        & ", broker_title = '" & Me!new_broker_title & "'" _
        & ", BrokerFirm = " & Me!new_broker_firm_id _
        & ", BrokerEmail = '" & Me!new_broker_email & "'" _
        & " WHERE id = " & broker_person_id
    conn.Execute str_sql, adExecuteNoRecords
    Me!cmd_add_update.Caption = "add"
    
    fix_rs.broker_persons_f
    
    With Forms(str_form)
        For i = 1 To arr_fields(0, 0)
            .Controls("new_" & arr_fields(i, 0)) = ""
            If arr_fields(i, global_vars.control_list_propertes.set_bg_to_white) = True Then
                .Controls("new_" & arr_fields(i, 0)).BackColor = colors.white
            End If
        Next i
    End With

outro:
    Load.call_stack = ""
    If Not rs Is Nothing Then
        If rs.State = 1 Then rs.Close
        Set rs = Nothing
    End If
    Exit Sub
err_handler:
    Dim err_object As cls_err_object
    Set err_object = New cls_err_object
    With err_object
        .call_stack = Load.call_stack
        .routine_name = "broker_persons_f.cmd_add_update.Click"
        .milestone = ""
        .params = "str_sql = " & str_sql
        .system_error_code = Err.Number
        .system_error_text = Err.Description
        .show_error_msg = True
        .send_error err_object
    End With
    
    GoTo outro

End Sub
Private Sub cmd_delete_Click()
    Load.check_conn_and_variables
    On Error GoTo err_handler
    If Load.is_debugging = True Then On Error GoTo 0
    
    Dim str_sql As String, rs As ADODB.Recordset, i As Integer

    Dim str_message As String, str_name As String, answer As Variant
    str_name = Nz(Me!personal_name, Me!broker_person_id)
    str_message = "You are about to delete " & str_name & ". Please confirm."
    
    answer = MsgBox(str_message, vbQuestion + vbYesNo, "Confirm deletion.")
    If answer = vbNo Then
        Exit Sub
    Else
        conn.Execute "UPDATE " & Load.sources.broker_persons_table & " SET is_deleted = 1 WHERE id = " & Me!broker_person_id
        fix_rs.broker_persons_f
    End If
    
outro:
    If Not rs Is Nothing Then
        If rs.State = 1 Then rs.Close
        Set rs = Nothing
    End If
    Exit Sub
err_handler:
    MsgBox "Something went wrong. Try once more. If it fails again, snip this to Christian." & vbNewLine & vbNewLine _
        & "Error number: " & Err.Number & vbNewLine _
        & "Error description: " & Err.Description & vbNewLine _
        & "Where: broker_persons_f.cmd_delete_Click" & vbNewLine _
        & "Parameters: me!id = " & Me!id & vbNewLine _
        & "App: Stella Admin Eur", , " Whoopsie daisies (like Hugh Grant in Notting Hill)"
    GoTo outro

End Sub

Private Function declare_fields_local()
    Dim arr_fields(0 To 50, 0 To 3) As Variant, i As Integer
    Dim field_name As Integer, is_mandatory As Integer, set_bg_to_white As Integer, field_in_view As Integer
    field_name = global_vars.control_list_propertes.field_name
    is_mandatory = global_vars.control_list_propertes.is_mandatory
    set_bg_to_white = global_vars.control_list_propertes.set_bg_to_white
    field_in_view = global_vars.control_list_propertes.field_in_recordset
    
    i = 1
    arr_fields(i, field_name) = "broker_person_id"
    arr_fields(i, is_mandatory) = False
    arr_fields(i, set_bg_to_white) = False
    arr_fields(i, field_in_view) = "broker_person_id"
    
    i = i + 1
    arr_fields(i, field_name) = "personal_name"
    arr_fields(i, is_mandatory) = True
    arr_fields(i, set_bg_to_white) = True
    arr_fields(i, field_in_view) = "personal_name"
    
    i = i + 1
    arr_fields(i, field_name) = "broker_email"
    arr_fields(i, is_mandatory) = True
    arr_fields(i, set_bg_to_white) = True
    arr_fields(i, field_in_view) = "BrokerEmail"
    
    i = i + 1
    arr_fields(i, field_name) = "broker_firm_id"
    arr_fields(i, is_mandatory) = True
    arr_fields(i, set_bg_to_white) = True
    arr_fields(i, field_in_view) = "BrokerFirm"
    
    i = i + 1
    arr_fields(i, field_name) = "broker_title"
    arr_fields(i, is_mandatory) = True
    arr_fields(i, set_bg_to_white) = True
    arr_fields(i, field_in_view) = "broker_title"
    
    
    arr_fields(0, 0) = i - 1
    declare_fields_local = arr_fields
End Function
