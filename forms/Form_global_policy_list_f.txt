Private Sub cmd_canada_Click()
    Const proc_name As String = "global_policy_list_f.cmd_canada.click"
    utilities.call_stack_add_item proc_name
    On Error GoTo err_handler
    If Load.is_debugging = True Then On Error GoTo 0
    
    Dim col As New Collection
    
    Load.check_conn_and_variables
    
    global_policy_list.update_navins_home_button global_policy_list.navins_canada
    col.Add global_policy_list.navins_canada
    utilities.paint_control Me.Name, col
    fix_rs.global_policy_list_f
outro:
    Exit Sub
err_handler:
    Central.err_handler proc_name, Err.Number, Err.Description, "", "", "", True
    Resume outro
End Sub

Private Sub comSector_AfterUpdate()
    Load.check_conn_and_variables
    If Me!comSector = 111 Then GoTo outro ' 111 is _All
        
    'populate com_sub_sector based on selection
    Dim str_sql As String, rs As ADODB.Recordset
    'remove existing items
    With Me!com_sub_sector
        Do While .ListCount > 0
            .RemoveItem (0)
        Loop
        .AddItem "-1;_All"
        str_sql = "SELECT sector_id id, sector_name menu_item FROM " & sources.sectors_table & " WHERE parent_sector_id = " & Me!comSector & " ORDER BY sector_name"
        Set rs = utilities.create_adodb_rs(conn, str_sql)
            Do While rs.EOF = False
                .AddItem rs!id & ";" & rs!menu_item
                rs.MoveNext
            Loop
        rs.Close
        .Value = -1
    End With
outro:
    If Not rs Is Nothing Then
        If rs.State = 1 Then rs.Close
        Set rs = Nothing
    End If
    Exit Sub
err_handler:
    MsgBox "Something went wrong. Try once more. If it fails again, snip this to Christian." & vbNewLine & vbNewLine _
        & "Error number: " & Err.Number & vbNewLine _
        & "Error description: " & Err.Description & vbNewLine _
        & "Where: deal_details_f.target_super_sector_id_AfterUpdate()" & vbNewLine _
        & "Parameters: me!dela_id = " & Me!deal_id & vbNewLine _
        & "App: Stella UW Eur", , " Whoopsie daisies (like Hugh Grant in Notting Hill)"
    GoTo outro
End Sub

Private Sub cmdAllNone_Click()
    Load.check_conn_and_variables
    Dim navins_home As cls_field
    If global_policy_list.navins_denmark.field_value = True Then
        'turn all off
        For Each navins_home In global_policy_list.col_navins_homes
            With navins_home
                Debug.Print .field_name
                .field_value = False
                .font_color = Load.colors.black
                .field_bg_color = Load.colors.inactive_navins_home
                .field_caption = Split(.field_caption, ":")(0) & ": off"
            End With
        Next navins_home
    Else
        'turn all on
        For Each navins_home In global_policy_list.col_navins_homes
            With navins_home
                .field_value = True
                .font_color = Load.colors.white
                .field_bg_color = Load.colors.active_navins_home
                .field_caption = Split(.field_caption, ":")(0) & ": on"
            End With
        Next navins_home
    End If
    utilities.paint_control Me.Name, global_policy_list.col_navins_homes
End Sub

Private Sub FixCountryColors(ByVal strCountry As String)
    Load.check_conn_and_variables
    If Controls(strCountry).BackColor = Load.colors.inactive_filter_bg Then
        Controls(strCountry).BackColor = colors.active_filter_bg
        Controls(strCountry).ForeColor = colors.white
    Else
        Controls(strCountry).BackColor = colors.inactive_filter_bg
        Controls(strCountry).ForeColor = colors.black
    End If
End Sub
Private Sub fix_deal_list_order(ByVal data_field As String)
    Load.check_conn_and_variables
    Dim str_sql() As String
    str_sql = Split(Me.RecordSource, "WHERE")
    str_sql = Split(str_sql(1), "ORDER")
    Dim str_direction As String
    Dim str_control As String
    str_control = "sort_" & data_field
    If Me.Controls(str_control).Caption = "<<<" Then
        Me.Controls(str_control).Caption = ">>>"
        str_direction = "ASC"
    Else
        Me.Controls(str_control).Caption = "<<<"
        str_direction = "DESC"
    End If
    open_forms.deal_list_f
    Central.update_deal_list_f str_sql(0) & " ORDER BY " & data_field & " " & str_direction
End Sub
